## 基本查找问题
#### LC.704 Binary Search
二分查找模板题
{% highlight C++ %}
int search(vector<int>& nums, int target) {
    int i = 0, j = nums.size() - 1;
    while(i + 1 < j)
    {
        int mid = i + (j - i) / 2;
        if(nums[mid] == target) return mid;
        else if(nums[mid] > target) j = mid;
        else i = mid;
    }
    if(nums[i] == target) return i;
    if(nums[j] == target) return j;
    return -1;
}
